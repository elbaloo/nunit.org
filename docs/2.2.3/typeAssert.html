TypeAssert

The TypeAssert class is contained in the nunit.framework.extensions assembly and provides a number of methods that are useful when asserting on type relationships.

TypeAssert.IsType( string expected, string actual );
TypeAssert.IsType( string expected, string actual, string message );
TypeAssert.IsType( string expected, string actual,
                string message, params object[] args );

TypeAssert.IsAssignableFrom( string expected, string actual );
TypeAssert.IsAssignableFrom( string expected, string actual, 
                string message );
TypeAssert.IsAssignableFrom( string expected, string actual,
                string message, params object[] args );

TypeAssert.IsSubclassOf( string expected, string actual );
TypeAssert.IsSubclassOf( string expected, string actual, 
                string message );
TypeAssert.IsSubclassOf( string expected, string actual,
                string message, params object[] args );

TypeAssert.Implements( string expected, string actual );
TypeAssert.Implements( string expected, string actual, 
                string message );
TypeAssert.Implements( string expected, string actual,
                string message, params object[] args );
